
### **Step 3: Deploy with Kubernetes**
```yaml
# kubernetes/deployment.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: gringo-module-training
spec:
  replicas: 3
  selector:
    matchLabels:
      app: training-knowledge
  template:
    metadata:
      labels:
        app: training-knowledge
    spec:
      containers:
      - name: training-module
        image: gringo/training-knowledge:latest
        env:
        - name: CLIENT_ID
          value: "{{ client_id }}"
```
